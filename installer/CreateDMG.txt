# make a temporary dmg, attach, copy the .app, eject, and compress
hdiutil create -srcfolder BACApp.UI.app -volname "BACApp.UI" -fs HFS+ -format UDRW MyApp-temp.dmg
hdiutil attach MyApp-temp.dmg -mountpoint /Volumes/BACApp.UI
cp -R BACApp.UI.app /Volumes/BACApp.UI/
hdiutil detach /Volumes/BACApp.UI
hdiutil convert MyApp-temp.dmg -format UDBZ -o BACApp.UI.dmg
rm MyApp-temp.dmg



##clear quaranteed flag
#cd to folder when DMG downloaded
xattr -d com.apple.quarantine BACApp.UI.dmg
